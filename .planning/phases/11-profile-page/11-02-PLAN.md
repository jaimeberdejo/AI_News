**Optimistic display name update (PROF-01):**
```typescript
async function handleSaveName(newName: string) {
  const previousName = profile?.display_name ?? null
  setProfile(prev => prev ? { ...prev, display_name: newName } : prev)  // optimistic
  const res = await fetch('/api/profile', {
    method: 'PATCH',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ display_name: newName }),
  })
  if (!res.ok) {
    setProfile(prev => prev ? { ...prev, display_name: previousName } : prev)  // rollback
  }
}
```
