**`cropToSquare(file: File): Promise<Blob>` function** (define outside component):
```typescript
function cropToSquare(file: File): Promise<Blob> {
  return new Promise((resolve, reject) => {
    const img = new Image()
    img.onload = () => {
      const size = Math.min(img.width, img.height)
      const canvas = document.createElement('canvas')
      canvas.width = size
      canvas.height = size
      const ctx = canvas.getContext('2d')!
      const offsetX = (img.width - size) / 2
      const offsetY = (img.height - size) / 2
      ctx.drawImage(img, offsetX, offsetY, size, size, 0, 0, size, size)
      canvas.toBlob(
        blob => blob ? resolve(blob) : reject(new Error('crop failed')),
        'image/jpeg',
        0.9
      )
    }
    img.onerror = reject
    img.src = URL.createObjectURL(file)
  })
}
```

```typescript
async function handleConfirmCrop() {
  if (!cropBlob || !user) return
  setIsUploading(true)

  // 1. Import browser Supabase client (NOT the server client)
  //    Use: import { createClient } from '../lib/supabase/client'
  const supabase = createClient()

  // 2. Upload to Storage — upsert:true replaces existing avatar
  const path = `${user.id}/avatar.jpg`
  const { error: uploadError } = await supabase.storage
    .from('avatars')
    .upload(path, cropBlob, { contentType: 'image/jpeg', upsert: true })

  if (uploadError) {
    console.error('Avatar upload failed:', uploadError)
    setIsUploading(false)
    setCropBlob(null)
    setCropPreviewUrl(null)
    return
  }

  // 3. Get public URL (clean URL, no cache-bust in DB)
  const { data } = supabase.storage.from('avatars').getPublicUrl(path)
  const publicUrl = data.publicUrl

  // 4. Save to profiles table via API
  await fetch('/api/profile', {
    method: 'PATCH',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ avatar_url: publicUrl }),
  })

  // 5. Update local state — avatarVersion cache-busts the img src
  setProfile(prev => prev ? { ...prev, avatar_url: publicUrl } : prev)
  setAvatarVersion(Date.now())

  // 6. Cleanup
  setCropBlob(null)
  setCropPreviewUrl(null)
  setIsUploading(false)
}
```
