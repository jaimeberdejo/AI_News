    Create `pipeline/script.py` implementing `select_and_write(articles: list[Article]) -> tuple[str, list[Story]]`.

    **Key schema constraints from migration:**
    - `editions.edition_date` has UNIQUE constraint — use upsert/on_conflict to avoid duplicate editions on re-run
    - `videos.position` is `smallint NOT NULL CHECK (position BETWEEN 1 AND 5)` — max 5 videos, Groq must select 3-5 stories
    - `videos.status` starts at 'pending', set to 'generating' when script is written (before TTS call per SCRIPT-03)

    **Implementation:**

    ```python
    import json
    import logging
    from datetime import date
    from groq import Groq
    from pipeline.db import get_db
    from pipeline.models import Article, Story

    logger = logging.getLogger(__name__)
    _groq: Groq | None = None

    def _get_groq() -> Groq:
        global _groq
        if _groq is None:
            _groq = Groq()   # reads GROQ_API_KEY from env
        return _groq


    def _get_or_create_edition(db) -> str:
        """Upsert today's edition row, return its UUID. Raises if already published."""
        today = str(date.today())
        # Check for existing published edition — skip run if already done today
        existing = db.table("editions").select("id, status").eq("edition_date", today).execute()
        if existing.data:
            row = existing.data[0]
            if row["status"] == "published":
                raise RuntimeError(f"Edition for {today} already published. Skipping re-run.")
            return row["id"]
        # Create new edition row
        result = db.table("editions").insert({
            "edition_date": today,
            "status": "pending",
        }).execute()
        return result.data[0]["id"]


    def _select_stories(articles: list[Article]) -> list[dict]:
        """Call Groq to select 3-5 most important distinct financial stories. Returns list of {index, reason}."""
        articles_text = "\n".join(
            f"{i+1}. {a.title}: {a.summary[:200]}"
            for i, a in enumerate(articles)
        )
        client = _get_groq()
        response = client.chat.completions.create(
            model="llama-3.3-70b-versatile",
            messages=[
                {"role": "system", "content": (
                    "You are a financial news editor. Select the 3 to 5 most important, "
                    "distinct financial stories from the list. Prioritize stories with market impact. "
                    "Output ONLY valid JSON: "
                    '{\"stories\": [{\"index\": 1, \"reason\": \"brief reason\"}]}'
                    " where index is the 1-based article number. Select exactly 3 to 5 stories."
                )},
                {"role": "user", "content": articles_text},
            ],
            response_format={"type": "json_object"},
            temperature=0.2,
        )
        data = json.loads(response.choices[0].message.content)
        stories = data.get("stories", [])
        # Clamp to 3-5
        if len(stories) < 3:
            logger.warning("Groq selected fewer than 3 stories (%d). Using all.", len(stories))
        return stories[:5]  # never exceed 5 (DB constraint)


    def _write_script(article: Article) -> str:
        """Call Groq to write a 30-45 second financial influencer script for one article."""
        client = _get_groq()
        response = client.chat.completions.create(
            model="llama-3.3-70b-versatile",
            messages=[
                {"role": "system", "content": (
                    "You are a dynamic financial news narrator. Write a script for a 30-45 second "
                    "short-form video in an energetic, direct 'financial influencer' tone. "
                    "Target 75-115 words. Lead with the key fact. Use short sentences. "
                    "No hashtags, no emojis, no calls-to-action. Return ONLY the script text."
                )},
                {"role": "user", "content": f"Headline: {article.title}\n\nSummary: {article.summary}"},
            ],
            temperature=0.7,
            max_tokens=200,
        )
        return response.choices[0].message.content.strip()


    def select_and_write(articles: list[Article]) -> tuple[str, list[Story]]:
        """
        1. Upsert today's edition in DB.
        2. Call Groq to select 3-5 stories.
        3. For each selected story: write script, insert video row with status='generating' (SCRIPT-03).
        Returns (edition_id, list[Story]).
        """
        if not articles:
            raise RuntimeError("No articles to select from — ingest returned empty list.")

        db = get_db()
        edition_id = _get_or_create_edition(db)
        logger.info("Edition ID: %s", edition_id)

        selected = _select_stories(articles)
        if not selected:
            raise RuntimeError("Groq story selection returned no stories.")

        stories: list[Story] = []
        for position, sel in enumerate(selected, start=1):
            article_index = sel["index"] - 1  # 0-based
            if article_index < 0 or article_index >= len(articles):
                logger.warning("Groq returned out-of-range index %d, skipping", sel["index"])
                continue
            article = articles[article_index]
            script_text = _write_script(article)
            logger.info("Script written for position %d: %s", position, article.title[:50])

            # SCRIPT-03: insert video row BEFORE TTS call
            result = db.table("videos").insert({
                "edition_id": edition_id,
                "position": position,
                "headline": article.title,
                "script_text": script_text,
                "source_url": article.url,
                "status": "generating",
            }).execute()
            video_id = result.data[0]["id"]

            stories.append(Story(
                position=position,
                headline=article.title,
                script_text=script_text,
                source_url=article.url,
                db_video_id=video_id,
            ))

        return edition_id, stories
    ```
