**VideoItem.tsx** — Single video screen. Mark with `'use client'` (uses useRef, event handlers):

```tsx
'use client'

import { useRef } from 'react'
import { useVideoPlayer } from '../hooks/useVideoPlayer'
import type { Video } from '../hooks/useEdition'

interface VideoItemProps {
  video: Video
  isMuted: boolean
  onEnded?: () => void
}

export function VideoItem({ video, isMuted, onEnded }: VideoItemProps) {
  const { containerRef, videoRef } = useVideoPlayer(isMuted)

  return (
    <div ref={containerRef} className="feed-item">
      {/* Video element — all four iOS attributes required */}
      <video
        ref={videoRef}
        src={`${video.video_url}#t=0.001`}
        autoPlay
        muted
        playsInline
        preload="auto"
        loop={false}
        onEnded={onEnded}
        className="absolute inset-0 w-full h-full object-cover"
        style={{ display: 'block' }}
      />

      {/* Headline overlay — bottom of screen, respects safe-area-inset-bottom */}
      <div
        className="absolute bottom-0 left-0 right-0 px-4 py-3 text-white"
        style={{
          paddingBottom: 'calc(env(safe-area-inset-bottom) + 48px)',
          background: 'linear-gradient(to top, rgba(0,0,0,0.7) 0%, transparent 100%)',
        }}
      >
        <p className="text-sm font-semibold leading-snug line-clamp-2">{video.headline}</p>
      </div>
    </div>
  )
}
```

Important notes:
- `muted` attribute is set as JSX prop (renders as `muted=""` on the DOM element) — this is the initial state. The IntersectionObserver in useVideoPlayer will sync `.muted` from the `isMuted` prop dynamically.
- `#t=0.001` fragment forces iOS Safari to decode the first video frame as a poster, preventing a black rectangle before play starts.
- `object-cover` fills the container maintaining aspect ratio — correct for 9:16 portrait video (720x1280 pipeline output).
- Headline overlay uses `env(safe-area-inset-bottom)` to avoid the iOS home indicator.

```tsx
'use client'

import { useState } from 'react'
import type { Video } from '../hooks/useEdition'
import { VideoItem } from './VideoItem'
import { MuteButton } from './MuteButton'

interface VideoFeedProps {
  videos: Video[]
}

export function VideoFeed({ videos }: VideoFeedProps) {
  const [isMuted, setIsMuted] = useState(true)
  const [activeIndex, setActiveIndex] = useState(0)
  const [showEndCard, setShowEndCard] = useState(false)

  if (videos.length === 0) {
    return (
      <div className="flex items-center justify-center h-screen bg-black text-white text-center p-8">
        <div>
          <p className="text-xl font-semibold">No videos yet</p>
          <p className="text-sm text-gray-400 mt-2">Check back soon for today&apos;s briefing</p>
        </div>
      </div>
    )
  }

  function handleVideoEnded() {
    if (activeIndex < videos.length - 1) {
      setActiveIndex(prev => prev + 1)
    } else {
      setShowEndCard(true)
    }
  }

  return (
    <div className="relative">
      {/* Mute button — top-right corner, above all video items */}
      <MuteButton isMuted={isMuted} onToggle={() => setIsMuted(prev => !prev)} />

      {/* Scroll container */}
      <div className="feed-container">
        {videos.map((video, idx) => (
          <VideoItem
            key={video.id}
            video={video}
            isMuted={isMuted}
            onEnded={idx === activeIndex ? handleVideoEnded : undefined}
          />
        ))}
      </div>

      {/* EndCard placeholder — to be replaced in Plan 03 */}
      {showEndCard && (
        <div
          className="fixed inset-0 bg-black flex items-center justify-center text-white"
          style={{ zIndex: 50 }}
        >
          <p>You&apos;re up to date</p>
        </div>
      )}
    </div>
  )
}
```

**Create frontend/hooks/useVideoPlayer.ts.**

This hook gives each VideoItem its own IntersectionObserver. When 70%+ of the container is visible, the video plays. Below that threshold, it pauses.

```ts
'use client'

import { useEffect, useRef } from 'react'

export function useVideoPlayer(isMuted: boolean) {
  const containerRef = useRef<HTMLDivElement>(null)
  const videoRef = useRef<HTMLVideoElement>(null)

  useEffect(() => {
    const container = containerRef.current
    const video = videoRef.current
    if (!container || !video) return

    const observer = new IntersectionObserver(
      ([entry]) => {
        if (entry.isIntersecting) {
          // Sync muted state from parent before playing
          video.muted = isMuted
          video.play().catch(() => {
            // Autoplay blocked — silently handle (browser policy)
          })
        } else {
          video.pause()
        }
      },
      { threshold: 0.7 }
    )

    observer.observe(container)
    return () => observer.unobserve(container)
  }, [isMuted])
  // isMuted in deps: when parent toggles mute, observer re-attaches with current value.
  // For the currently-playing video, the synchronous handler in MuteButton also sets
  // .muted directly on the ref (Plan 03). This hook handles the "start playing" sync.

  return { containerRef, videoRef }
}
```

Key design decisions:
- `threshold: 0.7` — video must be 70% visible before playing (standard for vertical feed)
- `isMuted` in `useEffect` deps — when the parent unmutes, the observer re-runs with the new muted value so the NEXT intersection call uses the updated state
- The currently-playing video's `.muted` is set synchronously by the MuteButton handler in Plan 03 (not via this hook) — this hook handles new video entry

```tsx
        color: 'white',

        WebkitBackdropFilter: 'blur(4px)',
```
