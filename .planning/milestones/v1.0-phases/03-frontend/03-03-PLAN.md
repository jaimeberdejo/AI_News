## Critical iOS constraint
`video.muted = false` MUST be called synchronously within the `onClick`/`onTouchEnd` handler. Any async boundary (Promise, await, setState + useEffect, setTimeout) breaks iOS's user-gesture requirement and results in silent video despite the icon showing unmuted.

## Implementation

**Step 1 — Add a module-level muted flag to VideoFeed.tsx.**

This variable is read synchronously in IntersectionObserver callbacks when a new video enters view. It must be module-level (not React state) so the IntersectionObserver callback can read the current value without waiting for a React re-render.

Add this at the TOP of VideoFeed.tsx, outside the component function:
```ts
// Module-level flag: read synchronously in IntersectionObserver callbacks.
// DO NOT store this in React state — IntersectionObserver callbacks run outside
// React's render cycle and cannot read stale closure state reliably.
let globalMuted = true
```

Update MuteButton.tsx to accept `videoRefs` prop:

```tsx
interface MuteButtonProps {
  isMuted: boolean
  onToggle: (newMuted: boolean) => void
  videoRefs: React.MutableRefObject<Array<React.RefObject<HTMLVideoElement>>>
}

export function MuteButton({ isMuted, onToggle, videoRefs }: MuteButtonProps) {
  const handleClick = (e: React.MouseEvent) => {
    // CRITICAL: everything here is synchronous — no async, no await
    const newMuted = !isMuted
    // 1. Update module-level flag first (read by IntersectionObserver callbacks)
    globalMuted = newMuted  // Note: globalMuted is imported from VideoFeed module
    // 2. Set .muted on all current video elements synchronously
    videoRefs.current.forEach(ref => {
      if (ref.current) ref.current.muted = newMuted
    })
    // 3. Update React state for UI re-render (icon change)
    onToggle(newMuted)
  }

  return (
    <button onClick={handleClick} aria-label={isMuted ? 'Unmute' : 'Mute'} ...>
      ...
    </button>
  )
}
```

**Step 5 — Update useVideoPlayer.ts** to read `globalMuted` when starting playback so that when a new video enters view (via IntersectionObserver), it starts with the correct muted state even if React state hasn't propagated yet.

In useVideoPlayer.ts, import and use globalMuted in the observer callback:

The cleanest approach is to pass `isMuted` as a ref (not state) to the hook so the observer always has the current value:

```ts
// In useVideoPlayer.ts — use isMutedRef pattern
export function useVideoPlayer(isMuted: boolean) {
  const containerRef = useRef<HTMLDivElement>(null)
  const videoRef = useRef<HTMLVideoElement>(null)
  const isMutedRef = useRef(isMuted)

  // Keep ref in sync with prop
  useEffect(() => {
    isMutedRef.current = isMuted
  }, [isMuted])

  useEffect(() => {
    const container = containerRef.current
    const video = videoRef.current
    if (!container || !video) return

    const observer = new IntersectionObserver(
      ([entry]) => {
        if (entry.isIntersecting) {
          // Use ref value — always current, even if React state is stale
          video.muted = isMutedRef.current
          video.play().catch(() => {})
        } else {
          video.pause()
        }
      },
      { threshold: 0.7 }
    )

    observer.observe(container)
    return () => observer.unobserve(container)
  }, []) // Empty deps: observer is created once; reads isMutedRef.current dynamically

  return { containerRef, videoRef }
}
```

This ensures the observer doesn't need to be re-created on every muted state change, which was causing subtle re-attach timing issues. The `isMutedRef` always has the current value.

```tsx
export function VideoItem({ video, isMuted, isActive, preloadOnly, onEnded, videoRefOverride }: VideoItemProps) {

        loop={false}
        onEnded={onEnded}
        className={preloadOnly ? undefined : "absolute inset-0 w-full h-full object-cover"}
        style={{ display: 'block' }}
      />
      {!preloadOnly && (
```

Note: Preload items use `position: fixed; width: 1px; height: 1px; visibility: hidden` — a 1px on-screen element. Research shows iOS is more reliable at buffering elements it considers "on screen" vs purely invisible. This is the most reliable iOS preload workaround.

```tsx
'use client'

import { useEffect, useState } from 'react'

interface EndCardProps {
  onReplay: () => void
  currentEditionId: string | null
  onNewEdition: () => void
}

function getNextEditionMessage(): string {
  const hour = new Date().getHours()
  if (hour < 6) return 'Check back this morning'
  if (hour < 12) return 'Check back this afternoon'
  if (hour < 18) return 'Check back tonight'
  return 'Check back tomorrow morning'
}

export function EndCard({ onReplay, currentEditionId, onNewEdition }: EndCardProps) {
  const [checkMessage] = useState(getNextEditionMessage)

  // Silent new-edition detection: poll /api/today once on mount.
  // If a different (newer) edition is available, call onNewEdition() to silently
  // refresh the feed — user doesn't see a loading state, the feed just updates.
  useEffect(() => {
    let cancelled = false

    async function checkForNewEdition() {
      try {
        const res = await fetch('/api/today', { cache: 'no-store' })
        const data = await res.json()
        const latestId = data?.edition?.id ?? null

        if (!cancelled && latestId && latestId !== currentEditionId) {
          // A newer edition is available — silently refresh
          onNewEdition()
        }
      } catch {
        // Network error — do nothing, user stays on end card
      }
    }

    checkForNewEdition()
    return () => { cancelled = true }
  }, [currentEditionId, onNewEdition])

  return (
    <div
      style={{
        position: 'fixed',
        inset: 0,
        background: '#000',
        display: 'flex',
        flexDirection: 'column',
        alignItems: 'center',
        justifyContent: 'center',
        zIndex: 50,
        padding: '32px',
        paddingTop: 'calc(env(safe-area-inset-top) + 32px)',
        paddingBottom: 'calc(env(safe-area-inset-bottom) + 32px)',
      }}
    >
      {/* Checkmark icon */}
      <div
        style={{
          width: '64px',
          height: '64px',
          borderRadius: '50%',
          border: '2px solid rgba(255,255,255,0.3)',
          display: 'flex',
          alignItems: 'center',
          justifyContent: 'center',
          marginBottom: '24px',

      >
        <svg width="28" height="28" viewBox="0 0 24 24" fill="none" stroke="white" strokeWidth="2">
          <polyline points="20 6 9 17 4 12" />
        </svg>
      </div>

      {/* Main message */}
      <h1
        style={{
          color: 'white',
          fontSize: '1.5rem',
          fontWeight: '700',
          textAlign: 'center',
          marginBottom: '8px',
```
